import tkinter as tk from tkinter import messagebox class SimpleApp: def __init__(self, root): """Initialize the application with a Tkinter window.""" self.root = root self.root.title("Button Click Event Example") # Create a button self.button = tk.Button(root, text="Click Me!", command=self.on_button_click) self.button.pack(pady=20) def on_button_click(self): """Event handler for button click.""" messagebox.showinfo("Button Clicked", "You clicked the button!") if __name__ == "__main__": # Create the root window root = tk.Tk() # Create an instance of the SimpleApp app = SimpleApp(root) # Run the Tkinter event loop root.mainloop()
