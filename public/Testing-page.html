class Car: def __init__(self, make, model, year, color): self.make = make self.model = model self.year = year self.color = color self.mileage = 0 # Initialize mileage to 0 def start(self): print(f"The {self.color} {self.year} {self.make} {self.model} is starting.") def stop(self): print(f"The {self.color} {self.year} {self.make} {self.model} is stopping.") def drive(self, distance): if distance > 0: self.mileage += distance print(f"The {self.year} {self.make} {self.model} is driving for {distance} miles.") else: print("Distance must be greater than 0.") def display_info(self): print(f"Car Info: {self.year} {self.make} {self.model}, Color: {self.color}, Mileage: {self.mileage} miles") # Creating an object of the Car class my_car = Car(make="Toyota", model="Camry", year=2020, color="Red") # Using the object my_car.start() my_car.drive(50) my_car.drive(20) my_car.stop() my_car.display_info() The Red 2020 Toyota Camry is starting. The 2020 Toyota Camry is driving for 50 miles. The 2020 Toyota Camry is driving for 20 miles. The Red 2020 Toyota Camry is stopping. Car Info: 2020 Toyota Camry, Color: Red, Mileage: 70 miles
